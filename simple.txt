
    model{
    # PRIORS
    lambda ~ dunif(0.01, 10)
    eta ~ dunif(1e-6, 10)

    # COVARIANCE MATRIX
    for(i in 1:NI){
    for(c in i:NI){
    M.cov[i,c] <- eta * exp(-dist[i,c] / lambda)
    }}
    
    for(i in 1:NI){
    for(c in (i+1):NI){
    M.cov[c,i] <- M.cov[i,c]
    }}
    
    M.tau[1:NI,1:NI] <- inverse(M.cov[,])

    # LIKELIHOOD
    g[1:NI] ~ dmnorm(mu[1:NI], M.tau[,])

    #data# NI, dist, mu
    #monitor# lambda, eta, g
    }
